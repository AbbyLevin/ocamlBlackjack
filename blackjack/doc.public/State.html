type state = {
   players : Player.player list;
   house : Player.player;
   game_name : string;
} 
val init_game_state : (string * string) list -> int -> string -> state

init_game_state player_names returns a game state with as many players as are in player_names plus the house

val player_turn : Player.player -> state -> Player.player

player_turn returns a player with their hand updated based on how many times they hit

val play_turns : state -> Player.player list -> Player.player list

play_turns returns a player list with each player's hand updated based on how many times they decided to hit

val deal_cards : state -> Player.player list -> Player.player list

deal_cards returns a player list where each player's hand is updated to contain two cards randomly selected from the deck

val place_bets : state -> Player.player list -> Player.player list

place_bets returns a players list where every player's bets for the current round have been placed

val determine_balances : state -> Player.player list

determine_balances returns a players list with updated balances based on how every player did against the house

val start_round : state -> state

start_round starts a new round of blackjack and returns the state once the game is finished

val winners_list : (Player.player * int) list -> (Player.player * int) list

winners_list lst returns a list of all of the winners of a round represented by lst and their scores.

val output_multiple_winners : (Player.player * int) list -> string -> bool -> string

output_multiple_winners winners_list returns a string that contains the names of all of the winners separated by a semicolon.

val house_win : (Player.player * int) list -> bool

house_win player_sums determines whether the house won

val get_winner : (Player.player * int) list -> string

get_winner player_sums determines the winner(s) of a list of players and their scores player_sums.

val output_multiple_winners_game : (Player.player * int) list -> string -> string

output_multiple_winners_game winners_list balance prints the winners within winners_list and their balance.

val get_winner_game : (Player.player * int) list -> string

get_winner player_money determines the winner(s) of a list of players and their balances player_sums.

val get_player_sums : (Player.player * int) list ->
       Player.player list -> (Player.player * int) list

get_player_sums acc players returns a dictionary acc where the key is each player's name in players and the associated value is their score at the end of the round.

val get_player_money : (Player.player * int) list ->
       Player.player list -> (Player.player * int) list

get_player_money acc players returns a dictionary acc where the key is each player's name in players and the associated value is their balance at the end of the round.

val compare_players : 'a * int -> 'b * int -> int

compare_players x y compares x and y based on their scores.
